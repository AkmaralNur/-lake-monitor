#include #include #include #include #include #include #include #include #include #include #define ONE_WIRE_BUS 4 #define DHT_PIN 27 #define DHT_TYPE DHT22 #define TDS_PIN 26 #define LIGHT_PIN 34 #define SCREEN_WIDTH 128 #define SCREEN_HEIGHT 64 #define OLED_RESET -1 const char* ssid = "akmaral"; const char* password = "123456780"; const char* apiKey = "EJVBVY1FPMIBPQUX"; OneWire oneWire(ONE_WIRE_BUS); DallasTemperature sensors(&oneWire); DHT dht(DHT_PIN, DHT_TYPE); Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET); // Переменные float tempWater, tempAir, humidity, tdsValue, lightLux; float dewPoint, heatIndex, evaporation, thermalFlux; float comfortIndex, eutrophicationIndex, wqi; int lightRaw; unsigned long lastTime = 0; // ===== ТОЧНЫЕ ФОРМУЛЫ ===== // DS18B20 — температура воды с компенсацией float calcWaterTemp(float raw) { // DS18B20 точность ±0.5°C — добавляем калибровочную поправку float calibOffset = 0.0; // подстрой если нужно (например -0.3) return raw + calibOffset; } // DHT22 — точная температура воздуха float calcAirTemp(float raw) { // DHT22 точность ±0.5°C float calibOffset = 0.0; return raw + calibOffset; } // DHT22 — точная влажность с температурной компенсацией float calcHumidity(float rawHum, float rawTemp) { // Компенсация влажности по температуре (психрометрическая поправка) float correctedHum = rawHum + (rawTemp - 25.0) * 0.1; return constrain(correctedHum, 0, 100); } // TDS — точная формула с температурной компенсацией float calcTDS(int rawADC, float waterTemp) { // Перевод ADC в напряжение (ESP32 12-bit ADC, 3.3V) float voltage = rawADC * (3.3 / 4095.0); // Линейная компенсация АЦП ESP32 (ESP32 ADC нелинейный!) // Поправка для диапазона 0-3.3V if (voltage < 0.1) return 0; // Температурная компенсация TDS // Стандартная температура измерения TDS = 25°C // Коэффициент компенсации: 2% на градус float tempCoeff = 1.0 + 0.02 * (waterTemp - 25.0); float compensatedVoltage = voltage / tempCoeff; // Формула перевода напряжения в TDS (ppm) // Калибровочная кривая для датчика TDS V1.0 float tds = (133.42 * pow(compensatedVoltage, 3) - 255.86 * pow(compensatedVoltage, 2) + 857.39 * compensatedVoltage) * 0.5; return max(0.0f, tds); } // Фоторезистор — перевод в люксы float calcLux(int rawADC) { if (rawADC <= 0) return 0; // Формула для GL5528 фоторезистора // Resistance = (4095 - rawADC) / rawADC * 10000 (при делителе 10кОм) float resistance = (float)(4095 - rawADC) / rawADC * 10000.0; // Перевод сопротивления в люксы по характеристике GL5528 // Lux = 10^((log10(R/10000) - b) / m) — логарифмическая зависимость float lux = pow(10, (log10(resistance / 10000.0) * (-1.4) + 3.0)); return constrain(lux, 0, 100000); } // Точка росы (формула Августа-Ромера-Магнуса) float calcDewPoint(float temp, float hum) { float a = 17.27, b = 237.7; float alpha = ((a * temp) / (b + temp)) + log(hum / 100.0); return (b * alpha) / (a - alpha); } // Индекс теплового комфорта (Heat Index) float calcHeatIndex(float temp, float hum) { // Формула Ротча (точная для T > 27°C) if (temp < 27) return temp; float hi = -8.78469475556 + 1.61139411 * temp + 2.33854883889 * hum - 0.14611605 * temp * hum - 0.012308094 * pow(temp, 2) - 0.0164248277778 * pow(hum, 2) + 0.002211732 * pow(temp, 2) * hum + 0.00072546 * temp * pow(hum, 2) - 0.000003582 * pow(temp, 2) * pow(hum, 2); return hi; } // Тепловой поток (разница температур вода-воздух) float calcThermalFlux(float waterTemp, float airTemp) { // Положительное значение = озеро отдаёт тепло // Отрицательное = озеро поглощает тепло return waterTemp - airTemp; } // Интенсивность испарения (упрощённая формула Пенмана) float calcEvaporation(float waterTemp, float airTemp, float humidity, float lux) { float satPressure = 0.6108 * exp(17.27 * waterTemp / (waterTemp + 237.3)); float actPressure = satPressure * humidity / 100.0; float vpd = satPressure - actPressure; // дефицит давления пара float radiation = lux * 0.0079; // грубый перевод люкс -> Вт/м² float evap = (0.408 * radiation + vpd * 0.066) / (1 + 0.34 * 2.0); return max(0.0f, evap); // мм/день } // Индекс комфорта у воды float calcComfortIndex(float airTemp, float humidity, float waterTemp) { // 0-100, где 100 = идеально комфортно float tempScore = 100 - abs(airTemp - 22) * 5; // идеал 22°C float humScore = 100 - abs(humidity - 55) * 2; // идеал 55% float waterScore = 100 - abs(waterTemp - 22) * 5; // идеал 22°C return constrain((tempScore + humScore + waterScore) / 3, 0, 100); } // Индекс эвтрофикации (риск "смерти" озера) float calcEutrophication(float waterTemp, float tds, float lux) { // 0-100, где 100 = критический риск float tempRisk = constrain((waterTemp - 15) * 5, 0, 40); float tdsRisk = constrain(tds / 10, 0, 40); float lightRisk = constrain(lux / 1000, 0, 20); return tempRisk + tdsRisk + lightRisk; } // Точный WQI (Water Quality Index) по стандарту NSF float calcWQI(float waterTemp, float tds, float humidity) { // Температурная составляющая (норма 15-25°C) float tempQ; if (waterTemp >= 15 && waterTemp <= 25) tempQ = 100; else if (waterTemp < 15) tempQ = max(0.0f, 100 - (15 - waterTemp) * 8); else tempQ = max(0.0f, 100 - (waterTemp - 25) * 8); // TDS составляющая (норма до 500 ppm) float tdsQ; if (tds <= 300) tdsQ = 100; else if (tds <= 500) tdsQ = 100 - (tds - 300) * 0.25; else if (tds <= 1000) tdsQ = 50 - (tds - 500) * 0.08; else tdsQ = 10; tdsQ = max(0.0f, tdsQ); // Взвешенный WQI (температура важнее для воды) return (tempQ * 0.6 + tdsQ * 0.4); } // ===== OLED ===== void updateOLED() { display.clearDisplay(); display.setTextSize(1); display.setTextColor(WHITE); display.setCursor(0, 0); display.print("Water:"); display.print(tempWater, 1); display.println("C"); display.setCursor(0, 10); display.print("Air: "); display.print(tempAir, 1); display.println("C"); display.setCursor(0, 20); display.print("Hum: "); display.print(humidity, 1); display.println("%"); display.setCursor(0, 30); display.print("TDS: "); display.print(tdsValue, 0); display.println("ppm"); display.setCursor(0, 40); display.print("Dew: "); display.print(dewPoint, 1); display.println("C"); display.setCursor(0, 50); display.print("WQI: "); display.print(wqi, 0); display.println("%"); display.display(); } // ===== ОТПРАВКА В THINGSPEAK ===== void sendToThingSpeak() { if (WiFi.status() != WL_CONNECTED) return; WiFiClientSecure client; client.setInsecure(); HTTPClient http; char url[500]; sprintf(url, "https://api.thingspeak.com/update?api_key=%s" "&field1=%.1f" // Температура воды "&field2=%.0f" // TDS "&field3=%.1f" // Температура воздуха "&field4=%.1f" // Влажность "&field5=%.0f" // Освещённость (люкс) "&field6=%.0f" // WQI "&field7=%.1f" // Точка росы "&field8=%.1f", // Тепловой поток apiKey, tempWater, tdsValue, tempAir, humidity, lightLux, wqi, dewPoint, thermalFlux ); http.begin(client, url); http.setTimeout(10000); int code = http.GET(); Serial.println("ThingSpeak Code: " + String(code)); http.end(); } void setup() { pinMode(4, INPUT_PULLUP); Serial.begin(115200); sensors.begin(); dht.begin(); if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) { Serial.println("OLED not found!"); } else { display.clearDisplay(); display.setTextColor(WHITE); display.setCursor(0, 0); display.println("Connecting..."); display.display(); } WiFi.begin(ssid, password); Serial.print("Connecting to WiFi"); while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); } Serial.println("\nWiFi connected! IP: " + WiFi.localIP().toString()); display.clearDisplay(); display.setCursor(0, 0); display.println("WiFi OK!"); display.println(WiFi.localIP().toString()); display.display(); } void loop() { // 1. Читаем сырые данные sensors.requestTemperatures(); float rawWaterTemp = sensors.getTempCByIndex(0); float rawAirTemp = dht.readTemperature(); float rawHumidity = dht.readHumidity(); lightRaw = analogRead(LIGHT_PIN); int tdsRaw = analogRead(TDS_PIN); // 2. Применяем точные формулы tempWater = calcWaterTemp(rawWaterTemp); tempAir = calcAirTemp(rawAirTemp); humidity = calcHumidity(rawHumidity, rawAirTemp); tdsValue = calcTDS(tdsRaw, tempWater); lightLux = calcLux(lightRaw); // 3. Вычисляем производные показатели dewPoint = calcDewPoint(tempAir, humidity); heatIndex = calcHeatIndex(tempAir, humidity); thermalFlux = calcThermalFlux(tempWater, tempAir); evaporation = calcEvaporation(tempWater, tempAir, humidity, lightLux); comfortIndex = calcComfortIndex(tempAir, humidity, tempWater); eutrophicationIndex = calcEutrophication(tempWater, tdsValue, lightLux); wqi = calcWQI(tempWater, tdsValue, humidity); // 4. Выводим в Serial Monitor Serial.println("===== ДАННЫЕ ====="); Serial.printf("Температура воды: %.2f °C\n", tempWater); Serial.printf("Температура воздуха: %.2f °C\n", tempAir); Serial.printf("Влажность: %.2f %%\n", humidity); Serial.printf("TDS: %.1f ppm\n", tdsValue); Serial.printf("Освещённость: %.0f люкс\n", lightLux); Serial.println("===== РАСЧЁТНЫЕ ====="); Serial.printf("Точка росы: %.2f °C\n", dewPoint); Serial.printf("Индекс жары: %.2f °C\n", heatIndex); Serial.printf("Тепловой поток: %.2f °C\n", thermalFlux); Serial.printf("Испарение: %.2f мм/день\n", evaporation); Serial.printf("Комфорт у воды: %.1f / 100\n", comfortIndex); Serial.printf("Риск эвтроф.: %.1f / 100\n", eutrophicationIndex); Serial.printf("WQI: %.1f %%\n", wqi); Serial.println("=================="); // 5.
